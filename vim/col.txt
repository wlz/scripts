       INP_BILL_DETAIL.CHARGES,
       INP_BILL_DETAIL.COSTS,
       INP_BILL_DETAIL.ITEM_CLASS,
       INP_BILL_DETAIL.ITEM_NAME,
       INP_BILL_DETAIL.ITEM_SPEC,
       INP_BILL_DETAIL.UNITS,
       BILL_ITEM_CLASS_DICT.CLASS_NAME,
       INP_BILL_DETAIL.BILLING_DATE_TIME,      
       USERS.USER_NAME,
       DEPT_DICT.DEPT_NAME
  from INP_BILL_DETAIL, BILL_ITEM_CLASS_DICT, DEPT_DICT, USERS
 where BILL_ITEM_CLASS_DICT.CLASS_CODE(+) = INP_BILL_DETAIL.ITEM_CLASS
   and DEPT_DICT.DEPT_CODE(+) = INP_BILL_DETAIL.PERFORMED_BY
   and USERS.USER_ID(+) = INP_BILL_DETAIL.OPERATOR_NO
   and INP_BILL_DETAIL.PATIENT_ID = 'H400262'
   and INP_BILL_DETAIL.VISIT_ID = '1'
 order by INP_BILL_DETAIL.ITEM_CLASS, INP_BILL_DETAIL.ITEM_NAME
